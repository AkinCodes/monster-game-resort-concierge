Move 2 — Rewind to the safe bookmark (time machine)

When to use Move 2 Use this when you've made commits that went sideways and you want a clean slate. This rewinds your branch to the exact state of the safe-starting-point tag. All commits after that point are erased, and all files are restored. It's a full rewind.
🗑️ Oscar LOVES Move 3: The Nuclear Option! (Red — Total Reset) NOW we're talking! This is Oscar's FAVOURITE. Move 3 is the nuclear option — it doesn't just erase your scribbles or rewind the movie. def _make_key(func_name: str, args: tuple, kwargs: dict) -> str:
    parts = [func_name]
    for arg in args:
        parts.append(
            f"{type(arg).__name__}:{id(arg) if hasattr(arg, '__dict__') else repr(arg)}"
        )   # Objects use id() (memory address), primitives use repr()

def _reciprocal_rank_fusion(self, bm25_results, dense_results, k=60, bm25_weight=0.4):
    scores = {}
    # BM25 scores (40% weight)
    for rank, (idx, _) in enumerate(bm25_results):
        doc = self.corpus[idx]
        scores[doc] = scores.get(doc, 0) + bm25_weight * (1 / (k + rank))

    # Dense scores (60% weight)
    for rank, (doc, _) in enumerate(dense_results):
        scores[doc] = scores.get(doc, 0) + (1 - bm25_weight) * (1 / (k + rank))

    # Documents in BOTH lists get BOOSTED (they appear twice!)
    ranked = sorted(scores.items(), key=lambda x: x[1], reverse=True)
    return [{"text": doc, "score": score} for doc, score in ranked]

def _compute_context_overlap(self, response_text, contexts):
    response_tokens = _tokenize(response_text)  # {"vampire", "manor", "has", "a", "pool"}
    context_tokens = set()
    for ctx in contexts:
        context_tokens.update(_tokenize(ctx))     # {"vampire", "manor", "gothic", "luxury"}

    overlap = response_tokens & context_tokens    # {"vampire", "manor"} — intersection!
    return len(overlap) / len(response_tokens)    # 2/5 = 0.4

def _compute_semantic_similarity(self, response_text, contexts):
    model = self._get_model()  # Lazy-load SentenceTransformer (same as RAG!)
    if model is None: return 0.0

    resp_emb = model.encode([response_text])[0]  # 384-dim vector
    ctx_emb = model.encode(contexts)               # Array of 384-dim vectors

    similarities = []
    for emb in ctx_emb:
        cos_sim = np.dot(resp_emb, emb) / (np.linalg.norm(resp_emb) * np.linalg.norm(emb))
        similarities.append(cos_sim)   # Cosine similarity: -1 to 1

    return max(similarities)  # Best match wins
_compute_source_attribution() — sentence-level grounding 
def _compute_source_attribution(self, response_text, contexts):
    sentences = _split_sentences(response_text)  # Split response into sentences
    context_tokens = set()
    for ctx in contexts:
        context_tokens.update(_tokenize(ctx))

    grounded = 0
    for sentence in sentences:
        sent_tokens = _tokenize(sentence)
        overlap_ratio = len(sent_tokens & context_tokens) / len(sent_tokens)
        if overlap_ratio >= 0.3:  # At least 30% of sentence words in context
            grounded += 1

    return grounded / len(sentences)  # What fraction of sentences are grounded?
